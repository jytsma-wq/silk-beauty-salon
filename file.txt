PROMPT 5A: Scroll-Triggered Animations
INSTRUCTION: Create a reusable scroll-reveal wrapper component at src/components/ScrollReveal.tsx that uses Framer Motion to animate elements as they enter the viewport. This component wraps any child element and applies a subtle fade-up animation.
'use client'
import { motion } from 'framer-motion'
import { ReactNode } from 'react'

interface ScrollRevealProps { children: ReactNode; delay?: number; className?: string }

export default function ScrollReveal({ children, delay = 0, className = '' }: ScrollRevealProps) {
  return (
    <motion.div
      initial={{ opacity: 0, y: 30 }}
      whileInView={{ opacity: 1, y: 0 }}
      viewport={{ once: true, margin: '-80px' }}
      transition={{ duration: 0.6, delay, ease: 'easeOut' }}
      className={className}
    >
      {children}
    </motion.div>
  )
}
CRITICAL: The animation MUST be subtle: opacity 0 to 1, y 30px to 0. Duration 0.6s with easeOut. The viewport trigger fires once (once: true) with a -80px margin so it triggers slightly before the element is fully in view. The delay parameter allows staggered animations for grid items. Do NOT use spring physics, bounce, or any exaggerated motion. The Galderma site is calm and deliberate, not flashy.
